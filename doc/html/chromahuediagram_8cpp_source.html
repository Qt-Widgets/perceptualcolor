228 void ChromaHueDiagram::paintEvent(QPaintEvent* event)
229 {
230  // We do not paint directly on the widget, but on a QImage buffer first:
231  // Render anti-aliased looks better. But as Qt documentation says:
232  //
233  // “Renderhints are used to specify flags to QPainter that may or
234  // may not be respected by any given engine.”
235  //
236  // Painting here directly on the widget might lead to different
237  // anti-aliasing results depending on the underlying window system. This is
238  // especially problematic as anti-aliasing might shift or not a pixel to the
239  // left or to the right. So we paint on a QImage first. As QImage (at
240  // difference to QPixmap and a QWidget) is independant of native platform
241  // rendering, it guarantees identical anti-aliasing results on all
242  // platforms. Here the quote from QPainter class documentation:
243  //
244  // “To get the optimal rendering result using QPainter, you should use
245  // the platform independent QImage as paint device; i.e. using QImage
246  // will ensure that the result has an identical pixel representation
247  // on any platform.”
248  QImage paintBuffer(m_diameter, m_diameter, QImage::Format_ARGB32);
249  paintBuffer.fill(Qt::transparent);
250  QPainter painter(&paintBuffer);
251 
252  QPen pen;
253 
254  // Paint the background.
255  cmsCIELCh backgroundColor;
256  backgroundColor.L = 50;
257  backgroundColor.C = 0;
258  backgroundColor.h = 0;
259  QBrush brush(m_rgbColorSpace->colorRgbBound(backgroundColor));
260  painter.setBrush(brush);
261  pen.setStyle(Qt::PenStyle::NoPen);
262  painter.setPen(pen);
263  painter.setRenderHint(QPainter::Antialiasing, true);
264  painter.drawEllipse(
265  m_border, // x
266  m_border, // y
267  m_diameter - 2 * m_border, // width
268  m_diameter - 2 * m_border // height
269  );
270  painter.setRenderHint(QPainter::Antialiasing, false);
271  // Paint the diagram itself as available in the cache.
272  updateDiagramCache();
273  painter.drawImage(0, 0, m_diagramImage);
274 
275  // Paint a thin color wheel for better orientation
276  updateWheelCache();
277  painter.drawImage(
278  0,
279  0,
280  m_wheelImage
281  );
282 
283  // paint also an additional marker indicating the hue
284  if (m_mouseEventActive) {
285  qreal radius = m_diameter / static_cast<qreal>(2) - 2 * m_markerThickness;
286  // get widget coordinates for our marker
287  QPointF myMarkerInner = PolarPointF(
288  radius - 4 * m_markerThickness,
289  m_color.toLch().h
290  ).toCartesian();
291  QPointF myMarkerOuter = PolarPointF(
292  radius,
293  m_color.toLch().h
294  ).toCartesian();
295  myMarkerInner.ry() *= -1;
296  myMarkerOuter.ry() *= -1;
297  myMarkerInner += QPointF(m_diagramOffset, m_diagramOffset);
298  myMarkerOuter += QPointF(m_diagramOffset, m_diagramOffset);
299  // draw the line
300  pen = QPen();
301  pen.setWidth(m_markerThickness);
302  pen.setCapStyle(Qt::FlatCap);
303  pen.setColor(Qt::black);
304  painter.setPen(pen);
305  painter.setRenderHint(QPainter::Antialiasing, true);
306  painter.drawLine(myMarkerInner, myMarkerOuter);
307  }
308 
309  /* Paint a focus indicator.
310  *
311  * We could paint a focus indicator (round or rectangular) around the marker.
312  * Depending on the currently selected hue for the diagram, it looks ugly
313  * because the colors of focus indicator and diagram do not harmonize, or
314  * it is mostly invisible the the colors are similar. So this apporach does
315  * not work well.
316  *
317  * It seems better to paint a focus indicator for the whole widget.
318  * We could use the style primitives to paint a rectangular focus indicator
319  * around the whole widget:
320  * style()->drawPrimitive(QStyle::PE_FrameFocusRect, &option, &painter, this);
321  * However, this does not work well because the chroma-lightness diagram has
322  * usually a triangular shape. The style primitive, however, often paints
323  * just a line at the bottom of the widget. That does not look good. An
324  * alternative approach is that we paint ourself a focus indicator only
325  * on the left of the diagram (which is the place of black/grey/white,
326  * so the won't be any problems with non-harmonic colors). We can
327  * get the color from the current style:
328  * QColor hightlightColor = palette().color(QPalette::Highlight);
329  * QBrush highlightBrush = palette().highlight();
330  * Then we have to design the line that we want to display. It is better to
331  * do that ourself instead of relying on generic QStyle::PE_Frame or similar
332  * solutions as their result seems to be quite unpredictible accross various
333  * styles. So we use markerThickness as line width and paint it at the
334  * left-most possible position. As the border() property accomodates also to
335  * markerRadius, the distance of the focus line to the real diagram also
336  * does, which looks nice.
337  */
338  if (hasFocus()) {
339  painter.setRenderHint(QPainter::Antialiasing);
340  pen = QPen();
341  pen.setWidth(m_markerThickness);
342  pen.setColor(
343  palette().color(QPalette::Highlight)
344  );
345  painter.setPen(pen);
346  brush = QBrush(Qt::transparent);
347  painter.setBrush(brush);
348  painter.drawEllipse(
349  QPoint(m_diagramOffset + 1, m_diagramOffset + 1), // center
350  (m_diameter - m_markerThickness) / 2, // x radius
351  (m_diameter - m_markerThickness) / 2 // y radius
352  );
353  }
354 
355  // Paint the marker on-the-fly.
356  painter.setRenderHint(QPainter::Antialiasing);
357  QPoint imageCoordinates = currentImageCoordinates();
358  pen = QPen();
359  pen.setWidth(m_markerThickness);
360  int markerBackgroundLightness = m_color.toLch().L; // range: 0..100
361  if (markerBackgroundLightness >= 50) {
362  pen.setColor(Qt::black);
363  } else {
364  pen.setColor(Qt::white);
365  }
366  painter.setPen(pen);
367  brush = QBrush(Qt::transparent);
368  painter.setBrush(brush);
369  painter.drawEllipse(
370  imageCoordinates.x() - m_markerRadius,
371  imageCoordinates.y() - m_markerRadius,
372  2 * m_markerRadius + 1,
373  2 * m_markerRadius + 1
374  );
375  PolarPointF lineEndPolar = PolarPointF(
376  QPointF(
377  imageCoordinates.x() - m_diagramOffset,
378  (imageCoordinates.y() - m_diagramOffset) * (-1)
379  )
380  );
381  qreal newRadial = lineEndPolar.radial() - m_markerRadius - 1.5;
382  QPointF lineEndCartesian;
383  if (newRadial > 0) {
384  lineEndCartesian = PolarPointF(newRadial, lineEndPolar.angleDegree()).toCartesian();
385  lineEndCartesian.setY(lineEndCartesian.y() * (-1));
386  lineEndCartesian += QPointF(m_diagramOffset, m_diagramOffset);
387  painter.drawLine(
388  QPointF(m_diagramOffset, m_diagramOffset),
389  lineEndCartesian
390  );
391  }
392 
393  // Paint the buffer to the actual widget
394  QPainter(this).drawImage(0, 0, paintBuffer);
395 }

397 void ChromaHueDiagram::wheelEvent(QWheelEvent* event)
398 {
399  if (
400  /* Do nothing while mouse mouvement is tracked anyway. This would be confusing. */
401  (!m_mouseEventActive) &&
402  /* Only react on wheel events when its in the appropriate place */
403  // TODO Wouldn't it be better to accept wheel events in the whole gray circle, even if outside the gamut?

405  /* Only react on good old vertical wheels, and not on horizontal wheels */
406  (event->angleDelta().y() != 0)

409  lch.h += Helper::wheelSteps(event) * m_singleStepHue;
410  setColor(FullColorDescription(m_rgbColorSpace, lch, FullColorDescription::outOfGamutBehaviour::sacrifyChroma));
411  } else {
412  event->ignore();
413  }

430 void ChromaHueDiagram::keyPressEvent(QKeyEvent *event)
431 {
432  // TODO The choise of keys does not seem to be very intuitive
433  cmsCIELCh lch = m_color.toLch();
434  switch (event->key()) {
435  case Qt::Key_Up:
436  lch.C += m_singleStepChroma;
437  break;
438  case Qt::Key_Down:
439  lch.C -= m_singleStepChroma;
440  break;
441  case Qt::Key_Left:
442  lch.h += m_singleStepHue;
443  break;
444  case Qt::Key_Right:
445  lch.h -= m_singleStepHue;
446  break;
447  case Qt::Key_PageUp:
448  lch.C += m_pageStepChroma;
449  break;
450  case Qt::Key_PageDown:
451  lch.C -= m_pageStepChroma;
452  break;
453  case Qt::Key_Home:
454  lch.h += m_pageStepHue;
455  break;
456  case Qt::Key_End:
457  lch.h -= m_pageStepHue;
458  break;
459  default:
460  /* Quote from Qt documentation:
461  *
462  * If you reimplement this handler, it is very important that you call the base class
463  * implementation if you do not act upon the key.
464  *
465  * The default implementation closes popup widgets if the user presses the key sequence
466  * for QKeySequence::Cancel (typically the Escape key). Otherwise the event is ignored,
467  * so that the widget's parent can interpret it.
468  */
469  QWidget::keyPressEvent(event);
470  return;
471  }
472  // Here we reach only if the key has been recognized. If not, in the default branch of the
473  // switch statement, we would have passed the keyPressEvent yet to the parent and returned.
474  if (lch.C < 0) {
475  lch.C = 0;
476  }
477  setColor(FullColorDescription(m_rgbColorSpace, lch, FullColorDescription::outOfGamutBehaviour::sacrifyChroma));
478 }

581 void ChromaHueDiagram::resizeEvent(QResizeEvent* event)
582 {
583  // Find the smaller one of width and height
584  int newDiameter = qMin(size().width(), size().height());
585  // Make sure the diameter is an odd integer (so that center of the
586  // coordinate system is exactly at a pixel, and not between two
587  // pixels.
588  if ((newDiameter % 2) == 0) {
589  newDiameter -= 1;
590  }
591  if (newDiameter < 0) {
592  newDiameter = 0;
593  }
594  // Update the widget
595  if (newDiameter != m_diameter) {
596  m_diameter = newDiameter;
597  m_diagramOffset = (m_diameter - 1) / 2;
598  m_diagramCacheReady = false;
599  m_wheelCacheReady = false;
600  // As by Qt documentation: The widget will be erased and receive a paint event immediately after processing the resize event. No drawing need be (or should be) done inside this handler.
601  }
602 }

708 QImage ChromaHueDiagram::generateDiagramImage(
709  const RgbColorSpace *colorSpace,
710  const int imageSize,
711  const int maxChroma,
712  const qreal lightness,
713  const int border
714 )
715 {
716  int maxIndex = imageSize - 1;
717  // Test if image size is too small.
718  if (maxIndex < 1) {
719  // maxIndex must be at least >= 1 for our algorithm. If they are 0, this would crash (division by 0). // TODO how to solve this?
720  return QImage();
721  }
722 
723  // Setup
724  cmsCIELab lab; // uses cmsFloat64Number internally
725  int x;
726  int y;
727  QColor tempColor;
728  QImage tempImage = QImage(
729  QSize(imageSize, imageSize),
730  QImage::Format_ARGB32
731  );
732  tempImage.fill(Qt::transparent); // Initialize the image with transparency
733  const qreal scaleFactor = static_cast<qreal>(2 * maxChroma) / (imageSize - 2 * border);
734 
735  // Paint the gamut.
736  lab.L = lightness;
737  for (y = border; y <= maxIndex - border; ++y) {
738  lab.b = maxChroma - (y - border) * scaleFactor; // floating point division thanks to static_cast to cmsFloat64Number
739  for (x = border; x <= maxIndex - border; ++x) {
740  lab.a = (x - border) * scaleFactor - maxChroma;
741  tempColor = colorSpace->colorRgb(lab);
742  if (tempColor.isValid()) {
743  // The pixel is within the gamut
744  tempImage.setPixelColor(x, y, tempColor);
745  }
746  }
747  }
748 
749  QImage result = QImage(
750  QSize(imageSize, imageSize),
751  QImage::Format_ARGB32
752  );
753  result.fill(Qt::transparent);
754  // Cut of everything outside the circle
755  QPainter myPainter(&result);
756  myPainter.setRenderHint(QPainter::Antialiasing, true);
757  myPainter.setPen(QPen(Qt::NoPen));
758  myPainter.setBrush(QBrush(tempImage));
759  myPainter.drawEllipse(
760  border, // x
761  border, // y
762  imageSize - 2 * border, // width
763  imageSize - 2 * border // height
764  );
765 
766  return result;
767 }

783 void ChromaHueDiagram::updateDiagramCache()
784 {
785  if (m_diagramCacheReady) {
786  return;
787  }
788 
789  // Update QImage
