142 {

206 void SimpleColorWheel::wheelEvent(QWheelEvent *event)
207 {
208  // The step (mesured in degree) that the hue angle is changed when
209  // a mouse wheel event occures.
210  // TODO What is a reasonable value for this?
211  static constexpr qreal wheelStep = 5;
212  qreal radius = contentDiameter() / static_cast<qreal>(2) - border();
213  PolarPointF myPolarPoint = fromWidgetCoordinatesToWheelCoordinates(event->pos());
214  if (
215  /* Do nothing while mouse mouvement is tracked anyway. This would be confusing. */
216  (!m_mouseEventActive) &&
217  /* Only react on wheel events when its in the wheel ribbon or in the inner hole. */
218  (myPolarPoint.radial() <= radius) &&
219  /* Only react on good old vertical wheels, and not on horizontal wheels */
220  (event->angleDelta().y() != 0)
221  ) {
222  setHue(m_hue + Helper::wheelSteps(event) * wheelStep);
223  } else {
224  event->ignore();
225  }
226 }

242 void SimpleColorWheel::keyPressEvent(QKeyEvent *event)
243 {
244  constexpr qreal wheelStep = 5;
245  constexpr qreal bigWheelStep = 15;
246  switch (event->key()) {
247  case Qt::Key_Plus:
248  setHue(m_hue + wheelStep);
249  break;
250  case Qt::Key_Minus:
251  setHue(m_hue - wheelStep);
252  break;
253  case Qt::Key_Insert:
254  setHue(m_hue + bigWheelStep);
255  break;
256  case Qt::Key_Delete:
257  setHue(m_hue - bigWheelStep);
258  break;
259  default:
260  /* Quote from Qt documentation:
261  *
262  * If you reimplement this handler, it is very important that you call the base class
263  * implementation if you do not act upon the key.
264  *
265  * The default implementation closes popup widgets if the user presses the key sequence
266  * for QKeySequence::Cancel (typically the Escape key). Otherwise the event is ignored,
267  * so that the widget's parent can interpret it.
268  */
269  QWidget::keyPressEvent(event);
270  break;
271  }
272 }

296 void SimpleColorWheel::paintEvent(QPaintEvent* event)
297 {
298  // We do not paint directly on the widget, but on a QImage buffer first:
299  // Render anti-aliased looks better. But as Qt documentation says:
300  //
301  // “Renderhints are used to specify flags to QPainter that may or
302  // may not be respected by any given engine.”
303  //
304  // Painting here directly on the widget might lead to different
305  // anti-aliasing results depending on the underlying window system. This is
306  // especially problematic as anti-aliasing might shift or not a pixel to the
307  // left or to the right. So we paint on a QImage first. As QImage (at
308  // difference to QPixmap and a QWidget) is independant of native platform
309  // rendering, it guarantees identical anti-aliasing results on all
310  // platforms. Here the quote from QPainter class documentation:
311  //
312  // “To get the optimal rendering result using QPainter, you should use
313  // the platform independent QImage as paint device; i.e. using QImage
314  // will ensure that the result has an identical pixel representation
315  // on any platform.”
316  QImage paintBuffer(size(), QImage::Format_ARGB32);
317  paintBuffer.fill(Qt::transparent);
318  QPainter painter(&paintBuffer);
319 
320  // paint the wheel from the cache
321  if (!m_wheelImageReady) {
322  updateWheelImage();
323  }
324  painter.drawImage(0, 0, m_wheelImage);
325 
326  // paint the marker
327  qreal radius = contentDiameter() / static_cast<qreal>(2) - border();
328  // get widget coordinates for our marker
329  QPointF myMarkerInner = fromWheelCoordinatesToWidgetCoordinates(
330  PolarPointF(
331  radius - m_wheelThickness,
332  m_hue
333  )
334  );
335  QPointF myMarkerOuter = fromWheelCoordinatesToWidgetCoordinates(
336  PolarPointF(
337  radius,
338  m_hue
339  )
340  );
341  // draw the line
342  QPen pen;
343  pen.setWidth(m_markerThickness);
344  pen.setCapStyle(Qt::FlatCap);
345  pen.setColor(Qt::black);
346  painter.setPen(pen);
347  painter.setRenderHint(QPainter::Antialiasing, true);
348  painter.drawLine(myMarkerInner, myMarkerOuter);
349 
350  // Paint a focus indicator if the widget has the focus
351  if (hasFocus()) {
352  pen.setWidth(m_markerThickness);
353  pen.setColor(
354  palette().color(QPalette::Highlight)
355  );
356  painter.setPen(pen);
357  painter.drawEllipse(
358  m_markerThickness / static_cast<qreal>(2),
359  m_markerThickness / static_cast<qreal>(2),
360  contentDiameter() - m_markerThickness,
361  contentDiameter() - m_markerThickness
362  );
363  }
364 
365  // Paint the buffer to the actual widget
366  QPainter(this).drawImage(0, 0, paintBuffer);
367 }

577 QImage SimpleColorWheel::generateWheelImage(
578  RgbColorSpace *colorSpace,
579  const int outerDiameter,
580  const int border,
581  const int thickness,
582  const qreal lightness,
583  const qreal chroma)
584 {
585  if (outerDiameter <= 0) {
586  return QImage();
587  }
588 
589  // Firsts of all, generate a non-antialised, intermediate, color wheel, but
590  // with some pixels extra at the inner and outer side. The overlap defines
591  // an overlap for the wheel, so there are some more pixels that are drawn
592  // at the outer and at the inner border of the wheel, to allow later
593  // clipping with antialising
594  constexpr int overlap = 1;
595  PolarPointF polarCoordinates;
596  int x;
597  int y;
598  QColor rgbColor;
599  cmsCIELCh LCh; // uses cmsFloat64Number internally
600  int maxExtension = outerDiameter - 1; // maximum value for x index and y index
601  qreal center = maxExtension / static_cast<qreal>(2);
602  QImage rawWheel = QImage(QSize(outerDiameter, outerDiameter), QImage::Format_ARGB32);
603  // Because there may be out-of-gamut colors for some hue (depending on the given
604  // lightness and chroma value) which are drawn transparent, it is important to
605  // initialize this image with a transparent background.
606  rawWheel.fill(Qt::transparent);
607  LCh.L = lightness;
608  LCh.C = chroma;
609  // minimalRadial: Adding "+ 1" would reduce thw workload (less pixel to
610  // process) and still work mostly, but not completly. It creates sometimes
611  // artefacts in the antialiasing process. So we don't do that.
612  qreal minimumRadial = center - thickness - border - overlap;
613  qreal maximumRadial = center - border + overlap;
614  for (x = 0; x <= maxExtension; ++x) {
615  for (y = 0; y <= maxExtension; ++y) {
616  polarCoordinates = PolarPointF(QPoint(x - center, center - y));
617  if (Helper::inRange<qreal>(minimumRadial, polarCoordinates.radial(), maximumRadial)) {
618  // We are within the wheel
619  LCh.h = polarCoordinates.angleDegree();
620  rgbColor = colorSpace->colorRgb(LCh);
621  if (rgbColor.isValid())
622  {
623  rawWheel.setPixelColor(
624  x,
625  y,
626  rgbColor
627  );
628  }
629  }
630  }
631  }
632 
633  // construct our final QImage with transparent background
634  QImage finalWheel = QImage(QSize(outerDiameter, outerDiameter), QImage::Format_ARGB32);
635  finalWheel.fill(Qt::transparent);
636 
637  // paint an anti-alised circle with the raw (non-antialiased) color wheel as brush
638  QPainter myPainter(&finalWheel);
639  myPainter.setRenderHint(QPainter::Antialiasing, true);
640  myPainter.setPen(QPen(Qt::NoPen));
641  myPainter.setBrush(QBrush(rawWheel));
642  myPainter.drawEllipse(border, border, outerDiameter - 2 * border, outerDiameter - 2 * border);
643 
644  // set the inner circle of the wheel to antialised transparency
645  myPainter.setCompositionMode(QPainter::CompositionMode_DestinationOut);
646  myPainter.setRenderHint(QPainter::Antialiasing, true);
647  myPainter.setPen(QPen(Qt::NoPen));
648  myPainter.setBrush(QBrush(Qt::SolidPattern));
649  myPainter.drawEllipse(
650  thickness + border,
651  thickness + border,
652  outerDiameter - 2 * (thickness + border),
653  outerDiameter - 2 * (thickness + border)
654  );
655 
656  // return
657  return finalWheel;
658 }
