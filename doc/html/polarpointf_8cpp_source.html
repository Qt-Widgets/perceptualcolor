70 PolarPointF::PolarPointF(const QPointF cartesianCoordiantes)
71 {
72  m_radial = sqrt(
73  pow(cartesianCoordiantes.x(), 2) + pow(cartesianCoordiantes.y(), 2)
74  );
75  if (m_radial == 0) {
76  m_angleDegree = 0;
77  return;
78  }
79  if (cartesianCoordiantes.y() >= 0) {
80  m_angleDegree = qRadiansToDegrees(
81  acos(cartesianCoordiantes.x() / m_radial)
82  );
83  } else {
84  // M_PI is defined by QtMath (also on platforms that don't
85  // support it natively)
86  m_angleDegree = qRadiansToDegrees(
87  2 * (M_PI) - acos(cartesianCoordiantes.x() / m_radial)
88  );
89  }
90 }

151 qreal PolarPointF::normalizedAngleDegree(const qreal angleDegree)
152 {
153  qreal temp = fmod(angleDegree, 360);

155  temp += 360;
156  }
