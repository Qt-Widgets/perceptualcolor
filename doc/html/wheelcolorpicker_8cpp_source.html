88 QSize WheelColorPicker::scaleRectangleToDiagonal(const QSize oldRectangle, const qreal newDiagonal)
89 {
90  if (oldRectangle.isEmpty()) {
91  return QSize();
92  }
93  qreal ratioWidthPerHeight = static_cast<qreal>(oldRectangle.width()) / oldRectangle.height();
94  int newHeight = sqrt(pow(newDiagonal, 2) / (1 + pow(ratioWidthPerHeight, 2)));
95  int newWidth = newHeight * ratioWidthPerHeight;
96  return QSize(newWidth, newHeight);
97 }
