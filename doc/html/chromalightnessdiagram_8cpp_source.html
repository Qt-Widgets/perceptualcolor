230 void ChromaLightnessDiagram::paintEvent(QPaintEvent* event)
231 {
232  // We do not paint directly on the widget, but on a QImage buffer first:
233  // Render anti-aliased looks better. But as Qt documentation says:
234  //
235  // “Renderhints are used to specify flags to QPainter that may or
236  // may not be respected by any given engine.”
237  //
238  // Painting here directly on the widget might lead to different
239  // anti-aliasing results depending on the underlying window system. This is
240  // especially problematic as anti-aliasing might shift or not a pixel to the
241  // left or to the right. So we paint on a QImage first. As QImage (at
242  // difference to QPixmap and a QWidget) is independant of native platform
243  // rendering, it guarantees identical anti-aliasing results on all
244  // platforms. Here the quote from QPainter class documentation:
245  //
246  // “To get the optimal rendering result using QPainter, you should use
247  // the platform independent QImage as paint device; i.e. using QImage
248  // will ensure that the result has an identical pixel representation
249  // on any platform.”
250  QImage paintBuffer(size(), QImage::Format_ARGB32);
251  paintBuffer.fill(Qt::transparent);
252  QPainter painter(&paintBuffer);
253 
254  QPen pen;
255 
256  // Paint the diagram itself as available in the cache.
257  updateDiagramCache();
258  painter.drawImage(m_border, m_border, m_diagramImage);
259 
260  /* Paint a focus indicator.
261  *
262  * We could paint a focus indicator (round or rectangular) around the marker.
263  * Depending on the currently selected hue for the diagram, it looks ugly
264  * because the colors of focus indicator and diagram do not harmonize, or
265  * it is mostly invisible the the colors are similar. So this apporach does
266  * not work well.
267  *
268  * It seems better to paint a focus indicator for the whole widget.
269  * We could use the style primitives to paint a rectangular focus indicator
270  * around the whole widget:
271  * style()->drawPrimitive(QStyle::PE_FrameFocusRect, &option, &painter, this);
272  * However, this does not work well because the chroma-lightness diagram has
273  * usually a triangular shape. The style primitive, however, often paints
274  * just a line at the bottom of the widget. That does not look good. An
275  * alternative approach is that we paint ourself a focus indicator only
276  * on the left of the diagram (which is the place of black/grey/white,
277  * so the won't be any problems with non-harmonic colors). We can
278  * get the color from the current style:
279  * QColor hightlightColor = palette().color(QPalette::Highlight);
280  * QBrush highlightBrush = palette().highlight();
281  * Then we have to design the line that we want to display. It is better to
282  * do that ourself instead of relying on generic QStyle::PE_Frame or similar
283  * solutions as their result seems to be quite unpredictible accross various
284  * styles. So we use markerThickness as line width and paint it at the
285  * left-most possible position. As the border() property accomodates also to
286  * markerRadius, the distance of the focus line to the real diagram also
287  * does, which looks nice.
288  */
289  if (hasFocus()) {
290  pen.setWidth(m_markerThickness);
291  pen.setColor(
292  palette().color(QPalette::Highlight)
293  );
294  painter.setPen(pen);
295  painter.drawLine(
296  m_markerThickness / 2, // 0.5 is rounded down to 0.0
297  0 + m_border,
298  m_markerThickness / 2, // 0.5 is rounded down to 0.0
299  size().height() - m_border
300  );
301  }
302 
303  // Paint the marker on-the-fly.
304  // Render anti-aliased looks better. But as Qt documentation says: “Renderhints are used to
305  // specify flags to QPainter that may or may not be respected by any given engine.” Now, we
306  // are painting here directly on the widget, which might lead to different anti-aliasing
307  // results depending on the underlying window system. An alternative approach might be to
308  // do the rendering on a QImage first. As QImage (at difference to QPixmap and widgets) is
309  // independant of native platform rendering, it would garantee identical results on all
310  // platforms. But it seems a litte overkill, so we don't do that here. Anyway here the
311  // quote from QPainter class documentation:
312  //
313  // To get the optimal rendering result using QPainter, you should use the platform independent
314  // QImage as paint device; i.e. using QImage will ensure that the result has an identical pixel
315  // representation on any platform.
316  painter.setRenderHint(QPainter::Antialiasing);
317  QPoint imageCoordinates = currentImageCoordinates();
318  pen.setWidth(m_markerThickness);
319  int markerBackgroundLightness = m_color.toLch().L; // range: 0..100
320  if (markerBackgroundLightness >= 50) {
321  pen.setColor(Qt::black);
322  } else {
323  pen.setColor(Qt::white);
324  }
325  painter.setPen(pen);
326  painter.drawEllipse(
327  imageCoordinates.x() + m_border - m_markerRadius,
328  imageCoordinates.y() + m_border - m_markerRadius,
329  2 * m_markerRadius + 1,
330  2 * m_markerRadius + 1
331  );
332 
333  // Paint the buffer to the actual widget
334  QPainter(this).drawImage(0, 0, paintBuffer);
335 }

364 void ChromaLightnessDiagram::keyPressEvent(QKeyEvent *event)
365 {
366  QPoint newImageCoordinates = currentImageCoordinates();
367  updateDiagramCache();
368  switch (event->key()) {
369  case Qt::Key_Up:
370  if (imageCoordinatesInGamut(newImageCoordinates + QPoint(0, -1))) {
371  newImageCoordinates += QPoint(0, -1);
372  }
373  break;
374  case Qt::Key_Down:
375  if (imageCoordinatesInGamut(newImageCoordinates + QPoint(0, 1))) {
376  newImageCoordinates += QPoint(0, 1);
377  }
378  break;
379  case Qt::Key_Left:
380  if (imageCoordinatesInGamut(newImageCoordinates + QPoint(-1, 0))) {
381  newImageCoordinates += QPoint(-1, 0);
382  }
383  break;
384  case Qt::Key_Right:
385  if (imageCoordinatesInGamut(newImageCoordinates + QPoint(1, 0))) {
386  newImageCoordinates += QPoint(1, 0);
387  }
388  break;
389  case Qt::Key_PageUp:
390  newImageCoordinates.setY(0);
391  while (!imageCoordinatesInGamut(newImageCoordinates + QPoint(0, 1))) {
392  newImageCoordinates += QPoint(0, 1);
393  }
394  break;
395  case Qt::Key_PageDown:
396  newImageCoordinates.setY(m_diagramImage.height() - 1);
397  while (!imageCoordinatesInGamut(newImageCoordinates + QPoint(0, -1))) {
398  newImageCoordinates += QPoint(0, -1);
399  }
400  break;
401  case Qt::Key_Home:
402  newImageCoordinates.setX(0);
403  while (!imageCoordinatesInGamut(newImageCoordinates + QPoint(1, 0))) {
404  newImageCoordinates += QPoint(1, 0);
405  }
406  break;
407  case Qt::Key_End:
408  newImageCoordinates.setX(m_diagramImage.width() - 1);
409  while (!imageCoordinatesInGamut(newImageCoordinates + QPoint(-1, 0))) {
410  newImageCoordinates += QPoint(-1, 0);
411  }
412  break;
413  default:
414  /* Quote from Qt documentation:
415  *
416  * If you reimplement this handler, it is very important that you call the base class
417  * implementation if you do not act upon the key.
418  *
419  * The default implementation closes popup widgets if the user presses the key sequence
420  * for QKeySequence::Cancel (typically the Escape key). Otherwise the event is ignored,
421  * so that the widget's parent can interpret it.
422  */
423  QWidget::keyPressEvent(event);
424  return;
425  }
426  // Here we reach only if the key has been recognized. If not, in the default branch of the
427  // switch statement, we would have passed the keyPressEvent yet to the parent and returned.
428  // Set the new image coordinates (only takes effect when image coordinates are indeed different)
429  setImageCoordinates(newImageCoordinates);
430 }

661 QImage ChromaLightnessDiagram::diagramImage(
662  const qreal imageHue,
663  const QSize imageSize) const
664 {
665  cmsCIELCh LCh; // uses cmsFloat64Number internally
666  QColor rgbColor;
667  int x;
668  int y;
669  QImage temp_image = QImage(imageSize, QImage::Format_ARGB32);
670  const int maxHeight = imageSize.height() - 1;
671  const int maxWidth = imageSize.width() - 1;
672 
673  // Test if image size is too small.
674  if ((maxHeight < 1) || (maxWidth < 1)) {
675  // TODO How to react correctly here? Exception?
676  // maxHeight and maxWidth must be at least >= 1 for our algorithm. If they are 0, this would crash (division by 0).
677  return temp_image;
678  }
679 
680  // Initialize the image with transparency.
681  temp_image.fill(Qt::transparent);
682 
683  // Paint the gamut.
684  LCh.h = PolarPointF::normalizedAngleDegree(imageHue);
685  for (y = 0; y <= maxHeight; ++y) {
686  LCh.L = y * static_cast<cmsFloat64Number>(100) / maxHeight; // floating point division thanks to 100 which is a "cmsFloat64Number"
687  for (x = 0; x <= maxWidth; ++x) {
688  // Using the same scale as on the y axis. floating point
689  // division thanks to 100 which is a "cmsFloat64Number"
690  LCh.C = x * static_cast<cmsFloat64Number>(100) / maxHeight;
691  rgbColor = m_rgbColorSpace->colorRgb(LCh);
692  if (rgbColor.isValid()) {
693  // The pixel is within the gamut
694  temp_image.setPixelColor(
695  x,
696  maxHeight - y,
697  rgbColor
698  );
699  /* If color is out-of-gamut: We have chroma on the x axis and
700  * lightness on the y axis. We are drawing the pixmap line per
701  * line, so we go for given lightness from low chroma to high
702  * chroma. Because of the nature of most gamuts, if once in a
703  * line we have an out-of-gamut value, all other pixels that
704  * are more at the right will be out-of-gamut also. So we
705  * could optimize our code and break here. But as we are not
706  * sure about this (we do not know the gamut at compile time)
707  * for the moment we do not optimize the code.
708  */
709  }
710  }
711  }
712 
713  return temp_image;
714 }
