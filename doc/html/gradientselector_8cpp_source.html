84 qreal GradientSelector::fromWindowCoordinatesToFraction(QPoint windowCoordinates)
85 {
86  qreal temp;
87  if (m_orientation == Qt::Orientation::Vertical) {
88  temp = (size().height() - windowCoordinates.y()) / static_cast<qreal>(size().height());
89  } else {
90  if (layoutDirection() == Qt::LayoutDirection::LeftToRight) {
91  temp = windowCoordinates.x() / static_cast<qreal>(size().width());
92  } else {
93  temp = (size().width() - windowCoordinates.x()) / static_cast<qreal>(size().width());
94  }
95  }
96  return qBound<qreal>(0, temp, 1);
97 }

152 void GradientSelector::keyPressEvent(QKeyEvent* event)
153 {
154  switch (event->key()) {
155  case Qt::Key_Up:
156  case Qt::Key_Plus:
157  setFraction(m_fraction + m_singleStep);
158  break;
159  case Qt::Key_Down:
160  case Qt::Key_Minus:
161  setFraction(m_fraction - m_singleStep);
162  break;
163  case Qt::Key_Left:
164  if (layoutDirection() == Qt::LayoutDirection::LeftToRight) {
165  setFraction(m_fraction - m_singleStep);
166  } else {
167  setFraction(m_fraction + m_singleStep);
168  }
169  break;
170  case Qt::Key_Right:
171  if (layoutDirection() == Qt::LayoutDirection::LeftToRight) {
172  setFraction(m_fraction + m_singleStep);
173  } else {
174  setFraction(m_fraction - m_singleStep);
175  }
176  break;
177  case Qt::Key_PageUp:
178  setFraction(m_fraction + m_pageStep);
179  break;
180  case Qt::Key_PageDown:
181  setFraction(m_fraction - m_pageStep);
182  break;
183  case Qt::Key_Home:
184  setFraction(0);
185  break;
186  case Qt::Key_End:
187  setFraction(1);
188  break;
189  default:
190  /* Quote from Qt documentation:
191  *
192  * If you reimplement this handler, it is very important that you call the base class
193  * implementation if you do not act upon the key.
194  *
195  * The default implementation closes popup widgets if the user presses the key sequence
196  * for QKeySequence::Cancel (typically the Escape key). Otherwise the event is ignored,
197  * so that the widget's parent can interpret it.
198  */
199  QWidget::keyPressEvent(event);
200  }

223 void GradientSelector::paintEvent(QPaintEvent* event)
224 {
225  // We do not paint directly on the widget, but on a QImage buffer first:
226  // Render anti-aliased looks better. But as Qt documentation says:
227  //
228  // “Renderhints are used to specify flags to QPainter that may or
229  // may not be respected by any given engine.”
230  //
231  // Painting here directly on the widget might lead to different
232  // anti-aliasing results depending on the underlying window system. This is
233  // especially problematic as anti-aliasing might shift or not a pixel to the
234  // left or to the right. So we paint on a QImage first. As QImage (at
235  // difference to QPixmap and a QWidget) is independant of native platform
236  // rendering, it guarantees identical anti-aliasing results on all
237  // platforms. Here the quote from QPainter class documentation:
238  //
239  // “To get the optimal rendering result using QPainter, you should use
240  // the platform independent QImage as paint device; i.e. using QImage
241  // will ensure that the result has an identical pixel representation
242  // on any platform.”
243  QImage paintBuffer(size(), QImage::Format_ARGB32);
244  paintBuffer.fill(Qt::transparent);
245  QPainter painter(&paintBuffer);
246 
247  painter.setTransform(getTransform());
248  if (!m_gradientImageReady) {
249  updateGradientImage();
250  }
251  painter.drawImage(0, 0, m_gradientImage);
252 
253 
254  int actualLength;
255  if (m_orientation == Qt::Orientation::Vertical) {
256  actualLength = size().height();
257  } else {
258  actualLength = size().width();
259  }
260 
261  QPolygonF arrowPolygon;
262  const qreal cursorPosition = actualLength * m_fraction;
263  const qreal arrowSize = 6;
264  arrowPolygon
265  << QPointF(cursorPosition, arrowSize)
266  << QPointF(cursorPosition + arrowSize, 0)
267  << QPointF(cursorPosition - arrowSize, 0);
268  painter.setBrush(QBrush(Qt::black));
269  QPen pen(Qt::transparent);
270  pen.setWidth(0);
271  painter.setPen(pen);
272  painter.drawPolygon(arrowPolygon);
273  arrowPolygon = QPolygonF(); // re-initialize
274  arrowPolygon
275  << QPointF(cursorPosition, m_gradientThickness - arrowSize)
276  << QPointF(cursorPosition + arrowSize, m_gradientThickness)
277  << QPointF(cursorPosition - arrowSize, m_gradientThickness);
278  painter.setBrush(QBrush(Qt::white));
279  painter.drawPolygon(arrowPolygon);
280  if (hasFocus()) {
281  pen.setWidth(2);
282  pen.setColor(
283  palette().color(QPalette::Highlight)
284  );
285  painter.setPen(pen);
286  painter.drawLine(
287  cursorPosition + arrowSize + 1,
288  0,
289  cursorPosition + arrowSize + 1,
290  m_gradientThickness
291  );
292  painter.drawLine(
293  cursorPosition - arrowSize,
294  0,
295  cursorPosition - arrowSize,
296  m_gradientThickness
297  );
298  }
299 
300  // Paint the buffer to the actual widget
301  QPainter(this).drawImage(0, 0, paintBuffer);
302 }

343 QPair<cmsCIELCh, qreal> GradientSelector::intermediateColor(const cmsCIELCh &firstColor, const cmsCIELCh &secondColor, qreal fraction)
344 {
345  cmsCIELCh color;
346  qreal alpha;
347  color.L = firstColor.L + (secondColor.L - firstColor.L) * fraction;
348  color.C = firstColor.C + (secondColor.C - firstColor.C) * fraction;
349  color.h = firstColor.h + (secondColor.h - firstColor.h) * fraction;
350  alpha = m_firstColor.alpha() + (m_secondColor.alpha() - m_firstColor.alpha()) * fraction;
351  return QPair<cmsCIELCh, qreal>(color, alpha);
352 }

369 void GradientSelector::updateGradientImage()
370 {
371  int actualLength;
372  int i;
373  if (m_orientation == Qt::Orientation::Vertical) {
374  actualLength = size().height();
375  } else {
376  actualLength = size().width();
377  }
378  QImage temp(actualLength, 1, QImage::Format_ARGB32);
379  temp.fill(Qt::transparent); // Initialize the image with transparency.
380  cmsCIELCh firstColor = m_firstColor.toLch();
381  cmsCIELCh secondColor = m_secondColor.toLch();
382  if (qAbs(firstColor.h - secondColor.h) > 180) {
383  if (firstColor.h > secondColor.h) {
384  secondColor.h += 360;
385  } else {
386  secondColor.h -= 360;
387  }
388  }
389  QPair<cmsCIELCh, qreal> color;
390  FullColorDescription fullColor;
391  for (i = 0; i < actualLength; ++i) {
392  color = intermediateColor(firstColor, secondColor, i / static_cast<qreal>(actualLength));
393  // TODO the in-gamut test fails because of rounding errors for full-chroma-colors. How can we support ignore out-of-gamut colors? How should they be rendered? Not identical to transparent, right?
394  fullColor = FullColorDescription(
395  m_rgbColorSpace,
396  color.first,
397  FullColorDescription::outOfGamutBehaviour::preserve,
398  color.second
399  );
400  temp.setPixelColor(i, 0, fullColor.toRgbQColor());
401  }
402  QImage result = QImage(actualLength, m_gradientThickness, QImage::Format_ARGB32);
403  QPainter painter(&result);
404  painter.fillRect(0, 0, actualLength, m_gradientThickness, m_brush);
405  for (i = 0; i < m_gradientThickness; ++i) {
406  painter.drawImage(0, i, temp);
407  }
408  m_gradientImage = result;
409  m_gradientImageReady = true;
410 }
