25 AlphaSelector::AlphaSelector(RgbColorSpace *colorSpace, QWidget *parent) : QWidget(parent)
26 {

28  QHBoxLayout *layout = new QHBoxLayout();
29  layout->setContentsMargins(0, 0, 0, 0);
30  m_gradientSelector = new GradientSelector(m_rgbColorSpace);
31  m_gradientSelector->setOrientation(Qt::Orientation::Horizontal);
32  m_doubleSpinBox = new QDoubleSpinBox();
33  m_doubleSpinBox->setAlignment(Qt::AlignmentFlag::AlignRight);
34  layout->addWidget(m_gradientSelector);

37  connect(

43  connect(
44  m_doubleSpinBox,
45  QOverload<double>::of(&QDoubleSpinBox::valueChanged),
46  this,
47  &AlphaSelector::setAlphaFromRepresentationFormat
48  );
49  cmsCIELCh lch;
50  lch.L = Helper::LchBoundaries::defaultLightness;
51  lch.C = Helper::LchBoundaries::defaultChroma;
52  lch.h = Helper::LchBoundaries::defaultHue;
53  FullColorDescription temp = FullColorDescription(
54  m_rgbColorSpace,
55  lch,
56  FullColorDescription::outOfGamutBehaviour::preserve
57  );
58  /* setRepresentation does only do something if the new value is different from the old one.
59  * As we rely on it for initiailzation, we set it first to some other value, and second
60  * to the really desired default value. */
61  setRepresentation(NumerFormat::one); // first set to an unwanted value
62  setRepresentation(NumerFormat::percent); // second set to the really wanted value
64  setAlpha(0); // first set to an unwanted value
65  setAlpha(1); // second set to the really wanted value
66  /* Not necessary to do the same for color() because the default constructor of this object
67  * constructs an invalid color anyway, and because our new value is valid, it will be set
68  * correctly. */
69  setColor(temp);
70 }

88 void AlphaSelector::setRepresentation(AlphaSelector::NumerFormat newRepresentation)
89 {
90  if (m_representation == newRepresentation) {
91  return;
92  }
93  m_representation = newRepresentation;
94  switch (m_representation) {
95  case AlphaSelector::NumerFormat::one:
96  m_doubleSpinBox->setMinimum(0);
97  m_doubleSpinBox->setMaximum(1);
98  m_doubleSpinBox->setSuffix(QLatin1String());
99  m_doubleSpinBox->setDecimals(2);
100  m_doubleSpinBox->setValue(m_alpha);
101  break;
102  case AlphaSelector::NumerFormat::percent:
103  m_doubleSpinBox->setMinimum(0);
104  m_doubleSpinBox->setMaximum(100);
105  m_doubleSpinBox->setSuffix(QStringLiteral(u"%"));
106  m_doubleSpinBox->setDecimals(0);
107  m_doubleSpinBox->setValue(m_alpha * 100);
108  break;
109  case AlphaSelector::NumerFormat::twoHundredAndFiftyFive:
110  m_doubleSpinBox->setMinimum(0);
111  m_doubleSpinBox->setMaximum(255);
112  m_doubleSpinBox->setSuffix(QLatin1String());
113  m_doubleSpinBox->setDecimals(0);
114  m_doubleSpinBox->setValue(m_alpha * 255);
115  break;
116  default:
117  throw;
118  }
119 }
