58 FullColorDescription::FullColorDescription(RgbColorSpace *colorSpace, QColor color)
59 {
60  if (!color.isValid()) {
61  m_valid = false;
62  return;
63  }
64  switch (color.spec()) {
65  case QColor::Spec::Hsv:
66  m_hsvQColor = color;
67  m_rgbQColor = color.toRgb();
68  break;
69  case QColor::Spec::Rgb:
70  m_rgbQColor = color;
71  m_hsvQColor = color.toHsv();
72  break;
73  default:
74  m_rgbQColor = color.toRgb();
75  m_hsvQColor = color.toHsv();
76  break;
77  }
78  m_rgb.red = m_rgbQColor.redF();
79  m_rgb.green = m_rgbQColor.greenF();
80  m_rgb.blue = m_rgbQColor.blueF();
81  m_lab = colorSpace->colorLab(m_rgbQColor);
82  m_lch = Helper::toLch(m_lab);
83  m_alpha = color.alphaF();
84  m_valid = true;

140 void FullColorDescription::moveChromaIntoGamut(RgbColorSpace *colorSpace)
141 {
142  // Test special case: If we are yet in-gamut…
143  if (colorSpace->inGamut(m_lch)) {
144  return;
145  }
146 
147  // Now we know: We are out-of-gamut…
148  cmsCIELCh lowerChroma {m_lch.L, 0, m_lch.h};
149  cmsCIELCh upperChroma {m_lch};
150  cmsCIELCh candidate;
151  if (colorSpace->inGamut(lowerChroma)) {
152  // Now we know for sure that lowerChroma is in-gamut and upperChroma is out-of-gamut…
153  candidate = upperChroma;
154  while (upperChroma.C - lowerChroma.C > Helper::gamutPrecision) {
155  // Our test candidate is half the way between lowerChroma and upperChroma:
156  candidate.C = (
157  (lowerChroma.C + upperChroma.C) / 2
158  );
159  if (colorSpace->inGamut(candidate)) {
160  lowerChroma = candidate;
161  } else {
162  upperChroma = candidate;
163  }
164  }
165  m_lch = lowerChroma;
166  } else {
167  if (m_lch.L < colorSpace->blackpointL()) {
168  m_lch.L = colorSpace->blackpointL();
169  m_lch.C = 0;
170  } else {
171  if (m_lch.L > colorSpace->whitepointL()) {
172  m_lch.L = colorSpace->blackpointL();
173  m_lch.C = 0;
174  }
175  }
176  }
177 }
