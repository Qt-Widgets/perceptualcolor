57 void ColorPatch::paintEvent(QPaintEvent *event)
58 {
59  QFrame::paintEvent(event);
60 
61  // We do not paint directly on the widget, but on a QImage buffer first:
62  // Render anti-aliased looks better. But as Qt documentation says:
63  //
64  // “Renderhints are used to specify flags to QPainter that may or
65  // may not be respected by any given engine.”
66  //
67  // Painting here directly on the widget might lead to different
68  // anti-aliasing results depending on the underlying window system. This is
69  // especially problematic as anti-aliasing might shift or not a pixel to the
70  // left or to the right. So we paint on a QImage first. As QImage (at
71  // difference to QPixmap and a QWidget) is independant of native platform
72  // rendering, it guarantees identical anti-aliasing results on all
73  // platforms. Here the quote from QPainter class documentation:
74  //
75  // “To get the optimal rendering result using QPainter, you should use
76  // the platform independent QImage as paint device; i.e. using QImage
77  // will ensure that the result has an identical pixel representation
78  // on any platform.”
79  if (m_color.isValid()) {
80  QImage paintBuffer(size(), QImage::Format_ARGB32);
81  paintBuffer.fill(Qt::transparent);
82  QPainter painter(&paintBuffer);
83  if (m_color.alphaF() < 1) {
84  painter.fillRect(contentsRect(), m_brush);
85  }
86  painter.fillRect(contentsRect(), m_color);
87  // Paint the buffer to the actual widget
88  QPainter(this).drawImage(0, 0, paintBuffer);
89  }
90 }
