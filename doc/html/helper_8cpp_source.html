57  QImage transparencyBackground()
58  {
59  // The width and height of 12 px per square might be a good choise
60  // because 12 is a multiple of 2, 3 and 4, which might make scaling

117  QPoint nearestNeighborSearch(const QPoint originalPoint, const QImage &image) {
118  // test for special case: originalPoint itself is within the image and non-transparent
119  if (image.valid(originalPoint)) {
120  if (image.pixelColor(originalPoint).alpha() == 255) {
121  return originalPoint;
122  }
123  }
124 
125  // No special case. So we have to actually perfor a nearest-neighbor-search.
126  int x;
127  int y;
128  int currentBestX;
129  int currentBestY;
130  int currentBestDistanceSquare = std::numeric_limits<int>::max();
131  int temp;
132  for (x = 0; x < image.width(); x++) {
133  for (y = 0; y < image.height(); y++) {
134  if (image.pixelColor(x, y).alpha() == 255) {
135  temp = pow(originalPoint.x() - x, 2) + pow(originalPoint.y() - y, 2);
136  if (temp < currentBestDistanceSquare) {
137  currentBestX = x;
138  currentBestY = y;
139  currentBestDistanceSquare = temp;
140  }
141  }
142  }
143  }
144  if (currentBestDistanceSquare == std::numeric_limits<int>::max()) {
145  return QPoint(0, 0);
146  } else {
147  return QPoint(currentBestX, currentBestY);
148  }
149  }
