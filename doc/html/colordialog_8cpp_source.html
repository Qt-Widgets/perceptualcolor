101 void ColorDialog::setCurrentColor(const QColor& color)
102 {
103  QColor temp;
104  if (color.isValid()) {
105  temp = color;
106  } else {
107  // For invalid colors same behaviour as QColorDialog
108  temp = QColor(Qt::black);

135 void ColorDialog::setCurrentOpaqueFullColor(const FullColorDescription& color)
136 {
137  if (m_isColorChangeInProgress || (!color.isValid()) || (color == m_currentOpaqueFullColor)) {
138  return;
139  }
140  m_isColorChangeInProgress = true;
141  QColor oldQColor = currentColor(); // currentColor() provides correct alpha treatment
142  m_currentOpaqueFullColor = color;
143  updateOpaqueColorWidgets(m_currentOpaqueFullColor);
144  if (currentColor() != oldQColor) {
145  Q_EMIT currentColorChanged(currentColor());
146  }
147  m_isColorChangeInProgress = false;
148 }

154 void ColorDialog::updateOpaqueColorWidgets(const FullColorDescription &color)
155 {
156  QColor tempRgbQColor = color.toRgbQColor();
157  tempRgbQColor.setAlpha(255);
158  m_rgbRedSpinbox->setValue(tempRgbQColor.redF() * 255);
159  m_rgbGreenSpinbox->setValue(tempRgbQColor.greenF() * 255);
160  m_rgbBlueSpinbox->setValue(tempRgbQColor.blueF() * 255);
161  m_hsvHueSpinbox->setValue(color.toHsvQColor().hsvHueF() * 360);
162  m_hsvSaturationSpinbox->setValue(color.toHsvQColor().hsvSaturationF() * 255);
163  m_hsvValueSpinbox->setValue(color.toHsvQColor().valueF() * 255);
164  m_colorPatch->setColor(tempRgbQColor);
165  m_hlcLineEdit->setText(
166  QString(QStringLiteral(u"%1 %2 %3"))
167  .arg(color.toLch().h, 0, 'f', 0)
168  .arg(color.toLch().L, 0, 'f', 0)
169  .arg(color.toLch().C, 0, 'f', 0)
170  );
171  m_rgbLineEdit->setText(tempRgbQColor.name());
172  m_lchLightnessSelector->setFraction(color.toLch().L / static_cast<qreal>(100));
173  m_chromaHueDiagram->setColor(color);
174  m_wheelColorPicker->setCurrentColor(m_currentOpaqueFullColor);
175  m_alphaSelector->setColor(m_currentOpaqueFullColor);
176 }

384 void ColorDialog::readHlcNumericValues()
385 {
386  cmsCIELCh lch;
387  QStringList temp = m_hlcLineEdit->text().split(QStringLiteral(u" "));
388  if (temp.size() == 3) {
389  lch.h = temp.at(0).toInt();
390  lch.L = qMin(temp.at(1).toInt(), 100);
391  lch.C = temp.at(2).toInt();
392  setCurrentOpaqueFullColor(
393  FullColorDescription(m_rgbColorSpace, lch, FullColorDescription::outOfGamutBehaviour::sacrifyChroma)
394  );
395  } else {
396  updateOpaqueColorWidgets(m_currentOpaqueFullColor);
397  }
398 }
