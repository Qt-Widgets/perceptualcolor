40 RgbColorSpace::RgbColorSpace(QObject *parent) : QObject(parent)
41 {
42  // Create an ICC v4 profile object for the Lab color space.
43  // NULL means: Default white point (D50) // TODO Does this make sense? sRGB white point is D65!
44  cmsHPROFILE labProfileHandle = cmsCreateLab4Profile(NULL);
45  // Create an ICC profile object for the sRGB color space.
46  cmsHPROFILE rgbProfileHandle = cmsCreate_sRGBProfile();
47  m_description = getInformationFromProfile(rgbProfileHandle, cmsInfoDescription);
48  m_description = tr("sRGB"); // TODO Do this only if the build-in sRGB is used, not when an actual external ICC profile is used.
49  // Create the transforms
50  m_transformLabToRgbHandle = cmsCreateTransform(
51  labProfileHandle, // input profile handle
52  TYPE_Lab_DBL, // input buffer format
53  rgbProfileHandle, // output profile handle
54  TYPE_RGB_DBL, // output buffer format
55  INTENT_ABSOLUTE_COLORIMETRIC, // rendering intent
56  0 // flags
57  );
58  m_transformLabToRgb16Handle = cmsCreateTransform(
59  labProfileHandle, // input profile handle
60  TYPE_Lab_DBL, // input buffer format
61  rgbProfileHandle, // output profile handle
62  TYPE_RGB_16, // output buffer format
63  INTENT_ABSOLUTE_COLORIMETRIC, // rendering intent
64  0 // flags
65  );
66  m_transformRgbToLabHandle = cmsCreateTransform(
67  rgbProfileHandle, // input profile handle
68  TYPE_RGB_DBL, // input buffer format
69  labProfileHandle, // output profile handle
70  TYPE_Lab_DBL, // output buffer format
71  INTENT_ABSOLUTE_COLORIMETRIC, // rendering intent
72  0 // flags
73  );
74  // Close profile (free memory)
75  cmsCloseProfile(labProfileHandle);
76  cmsCloseProfile(rgbProfileHandle);
77 
78  // Now we know for sure that lowerChroma is in-gamut and upperChroma is out-of-gamut…
79  cmsCIELCh candidate;
80  candidate.L = 0;
81  candidate.C = 0;
82  candidate.h = 0;
83  while (!inGamut(candidate) && (candidate.L < 100)) {
84  candidate.L += Helper::gamutPrecision;
85  }
86  m_blackpointL = candidate.L;
87  candidate.L = 100;
88  while (!inGamut(candidate) && (candidate.L > 0)) {
89  candidate.L -= Helper::gamutPrecision;
90  }
91  m_whitepointL = candidate.L;
92  if (m_whitepointL <= m_blackpointL) {
93  qCritical() << "Unable to find blackpoint and whitepoint on gray axis.";
94  throw 0;
95  }
96 }

156 QColor RgbColorSpace::colorRgb(const cmsCIELab &Lab) const
157 {
158  QColor temp; // By default, without initialization this is an invalid color
159  Helper::cmsRGB rgb;
160  cmsDoTransform(m_transformLabToRgbHandle, &Lab, &rgb, 1); // convert exactly 1 value
161  if (Helper::inRange<cmsFloat64Number>(0, rgb.red, 1) &&
162  Helper::inRange<cmsFloat64Number>(0, rgb.green, 1) &&
163  Helper::inRange<cmsFloat64Number>(0, rgb.blue, 1)) {
164  // We are within the gamut
165  temp = QColor::fromRgbF(rgb.red, rgb.green, rgb.blue);
166  }
167  return temp;
168 }

184 Helper::cmsRGB RgbColorSpace::colorRgbBoundSimple(const cmsCIELab &Lab) const
185 {
186  cmsUInt16Number rgb_int[3];
187  cmsDoTransform(m_transformLabToRgb16Handle, &Lab, rgb_int, 1); // convert exactly 1 value
188  Helper::cmsRGB temp;
189  temp.red = rgb_int[0] / static_cast<qreal>(65535);
190  temp.green = rgb_int[1] / static_cast<qreal>(65535);
191  temp.blue = rgb_int[2] / static_cast<qreal>(65535);
192  return temp;
193 }

281 QString RgbColorSpace::getInformationFromProfile(cmsHPROFILE profileHandle, cmsInfoType infoType)
282 {
283  char languageCode[3] = "en"; // recommended default value following LittleCMS documentation
284  char countryCode[3] = "US"; // recommended default value following LittleCMS documentation
285  // Update languageCode and countryCode to actual values if possible
286  // TODO What is the result does not contain, as expected, 2 x 2 characters? Crash!
287  QStringList list = QLocale().name().split('_');
288  if (list.at(0).size() == 2) {
289  languageCode[0] = list.at(0).at(0).toLatin1();
290  languageCode[1] = list.at(0).at(1).toLatin1();
291  }
292  if (list.at(1).size() == 2) {
293  countryCode[0] = list.at(1).at(0).toLatin1();
294  countryCode[1] = list.at(1).at(1).toLatin1();
295  }
296 
297  // Get the size of the buffer that cmsGetProfileInfo needs to return a value
298  cmsUInt32Number resultLength = cmsGetProfileInfo(
299  profileHandle, // profile in which we search
300  infoType, // the type of information we search
301  languageCode, // the preferred language in which we want to get the information
302  countryCode, // the preferred country in which we want to get the information
303  NULL, // do not actually return the information, just calculate the required buffer size
304  0 // do not actually return the information, just calculate the required buffer size
305  );
306  // For the actual buffer, we add +1 which will later serve to guarantee zero-terminated string
307  cmsUInt32Number bufferLength = resultLength + 1;
308  // Allocate the buffer
309  wchar_t *buffer = new wchar_t [bufferLength];
310  // Initialize the buffer with NULL
311  for (int i = 0; i < bufferLength - 1; ++i) {
312  *(buffer + i) = 0;
313  }
314  // Get the actual description
315  cmsGetProfileInfo(
316  profileHandle, // profile in which we search
317  infoType, // the type of information we search
318  languageCode, // the preferred language in which we want to get the information
319  countryCode, // the preferred country in which we want to get the information
320  buffer, // the buffer into which the requested information will be writen
321  resultLength // the buffer size as previously calculated
322  );
323  /* QString::fromWCharArray allows to specify the number of elements to read.
324  * But cmsGetProfileInfo returns often strings that are smaller than the
325  * previewed buffer size. So it's better to read-in the buffer as
326  * null-terminated buffer.
327  */
328  // Make absolutely sure the buffer is null-terminated by marking its last
329  // element (the one that was the +1 "extra" element) as NULL.
330  *(buffer + (bufferLength - 1)) = 0;
331  // Read-in from the buffer
332  QString result = QString::fromWCharArray(
333  buffer, // read from the buffer
334  -1 // read until the first NULL element
335  );
336  // Free allocated memory of the buffer
337  delete [] buffer;
338  // Return
339  return result;
340 }
